Fonction sinusoïdale avec différentes fréquences

    
# Import matplotlib and numpy
import matplotlib.pyplot as plt
import numpy as np
# Import ipywidgets for the slider and the interact function
import ipywidgets as widgets
from ipywidgets import interact
# Define a function to plot the sine function with a given frequency
def plot_sine(f):
    # Create an array of x values from 0 to 2 pi
    x = np.linspace(0, 2 * np.pi, 100)
    # Compute the sine of x with the given frequency
    y = np.sin(f * x)
    # Plot the sine curve
    plt.plot(x, y)
    # Show the plot
    plt.show()
# Create a slider widget for the frequency
slider = widgets.FloatSlider(min=1, max=10, step=0.1, value=1)
# Use the interact function to link the slider and the plot_sine function
interact(plot_sine, f=slider)
